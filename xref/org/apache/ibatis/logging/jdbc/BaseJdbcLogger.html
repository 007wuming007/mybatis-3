29  /**
30   * Base class for proxies to do logging
31   * 
32   * @author Clinton Begin
33   * @author Eduardo Macarron
34   */
35  public abstract class BaseJdbcLogger {
36  
37    protected static final Set<String> SET_METHODS = new HashSet<String>();
38    protected static final Set<String> EXECUTE_METHODS = new HashSet<String>();
39  
40    private Map<Object, Object> columnMap = new HashMap<Object, Object>();
41  
42    private List<Object> columnNames = new ArrayList<Object>();
43    private List<Object> columnValues = new ArrayList<Object>();
44  
45    protected Log statementLog;
46    protected int queryStack;
47  
48    /*
49     * Default constructor
50     */
51    public BaseJdbcLogger(Log log, int queryStack) {
52      this.statementLog = log;
53      if (queryStack == 0) queryStack = 1;
54      this.queryStack = queryStack;
55    }
56  
57    static {
58      SET_METHODS.add("setString");
59      SET_METHODS.add("setInt");
60      SET_METHODS.add("setByte");
61      SET_METHODS.add("setShort");
62      SET_METHODS.add("setLong");
63      SET_METHODS.add("setDouble");
64      SET_METHODS.add("setFloat");
65      SET_METHODS.add("setTimestamp");
66      SET_METHODS.add("setDate");
67      SET_METHODS.add("setTime");
68      SET_METHODS.add("setArray");
69      SET_METHODS.add("setBigDecimal");
70      SET_METHODS.add("setAsciiStream");
71      SET_METHODS.add("setBinaryStream");
72      SET_METHODS.add("setBlob");
73      SET_METHODS.add("setBoolean");
74      SET_METHODS.add("setBytes");
75      SET_METHODS.add("setCharacterStream");
76      SET_METHODS.add("setClob");
77      SET_METHODS.add("setObject");
78      SET_METHODS.add("setNull");
79  
80      EXECUTE_METHODS.add("execute");
81      EXECUTE_METHODS.add("executeUpdate");
82      EXECUTE_METHODS.add("executeQuery");
83      EXECUTE_METHODS.add("addBatch");
84    }
85  
86    protected void setColumn(Object key, Object value) {
87      columnMap.put(key, value);
88      columnNames.add(key);
89      columnValues.add(value);
90    }
91  
92    protected Object getColumn(Object key) {
93      return columnMap.get(key);
94    }
95  
96    protected String getParameterValueString() {
97      List<Object> typeList = new ArrayList<Object>(columnValues.size());
98      for (Object value : columnValues) {
99        if (value == null) {
100         typeList.add("null");
101       } else {
102         typeList.add(value + "(" + value.getClass().getSimpleName() + ")");
103       }
104     }
105     final String parameters = typeList.toString();
106     return parameters.substring(1, parameters.length() - 1);
107   }
108 
109   protected String getColumnString() {
110     return columnNames.toString();
111   }
112 
113   protected void clearColumnInfo() {
114     columnMap.clear();
115     columnNames.clear();
116     columnValues.clear();
117   }
118 
119   protected String removeBreakingWhitespace(String original) {
120     StringTokenizer whitespaceStripper = new StringTokenizer(original);
121     StringBuilder builder = new StringBuilder();
122     while (whitespaceStripper.hasMoreTokens()) {
123       builder.append(whitespaceStripper.nextToken());
124       builder.append(" ");
125     }
126     return builder.toString();
127   }
128 
129   protected boolean isDebugEnabled() {
130     return statementLog.isDebugEnabled();
131   }
132 
133   protected boolean isTraceEnabled() {
134     return statementLog.isTraceEnabled();
135   }
136 
137   protected void debug(String text, boolean input) {
138     if (statementLog.isDebugEnabled()) {
139       statementLog.debug(prefix(input) + text);
140     }
141   }
142 
143   protected void trace(String text, boolean input) {
144     if (statementLog.isTraceEnabled()) {
145       statementLog.trace(prefix(input) + text);
146     }
147   }
148 
149   private String prefix(boolean isInput) {
150     char[] buffer = new char[queryStack * 2 + 2];
151     Arrays.fill(buffer, '=');
152     buffer[queryStack * 2 + 1] = ' ';
153     if (isInput) {
154       buffer[queryStack * 2] = '>';
155     } else {
156       buffer[0] = '<';
157     }
158     return new String(buffer);
159   }
