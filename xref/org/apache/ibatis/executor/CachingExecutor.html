16  package org.apache.ibatis.executor;

18  import java.sql.SQLException;
19  import java.util.List;
20

34  public class CachingExecutor implements Executor {
35  
36    private Executor delegate;
37    private boolean autoCommit; // issue #573. No need to call commit() on autoCommit sessions
38    private TransactionalCacheManager tcm = new TransactionalCacheManager();
39  
40    private boolean dirty;
41  
42    public CachingExecutor(Executor delegate) {
43      this(delegate, false);
44    }
45  
46    public CachingExecutor(Executor delegate, boolean autoCommit) {
47      this.delegate = delegate;
48      this.autoCommit = autoCommit;
49    }
50  
51    public Transaction getTransaction() {
52      return delegate.getTransaction();
53    }
54  
55    public void close(boolean forceRollback) {
56      try {
57        //issue #499. Unresolved session handling
58        //issue #573. Autocommit sessions should commit
59        if (dirty && !autoCommit) { 
60          tcm.rollback();
61        } else {
62          tcm.commit();
63        }
64      } finally {
65        delegate.close(forceRollback);
66      }
67    }
68  
69    public boolean isClosed() {
70      return delegate.isClosed();
71    }
72  
73    public int update(MappedStatement ms, Object parameterObject) throws SQLException {
74      flushCacheIfRequired(ms);
75      return delegate.update(ms, parameterObject);
76    }
77  
78    public <E> List<E> query(MappedStatement ms, Object parameterObject, RowBounds rowBounds, ResultHandler resultHandler) throws SQLException {
79      BoundSql boundSql = ms.getBoundSql(parameterObject);
80      CacheKey key = createCacheKey(ms, parameterObject, rowBounds, boundSql);
81      return query(ms, parameterObject, rowBounds, resultHandler, key, boundSql);
82    }
83  
84    public <E> List<E> query(MappedStatement ms, Object parameterObject, RowBounds rowBounds, ResultHandler resultHandler, CacheKey key, BoundSql boundSql) throws SQLException {
85      Cache cache = ms.getCache();
86      if (cache != null) {
87        flushCacheIfRequired(ms);
88        if (ms.isUseCache() && resultHandler == null) { 
89          ensureNoOutParams(ms, parameterObject, boundSql);
90          if (!dirty) {
91            cache.getReadWriteLock().readLock().lock();
92            try {
93              @SuppressWarnings("unchecked")
94              List<E> cachedList = (List<E>) cache.getObject(key);
95              if (cachedList != null) return cachedList;
96            } finally {
97              cache.getReadWriteLock().readLock().unlock();
98            }
99          }
100         List<E> list = delegate.<E> query(ms, parameterObject, rowBounds, resultHandler, key, boundSql);
101         tcm.putObject(cache, key, list); // issue #578. Query must be not synchronized to prevent deadlocks
102         return list;
103       }
104     }
105     return delegate.<E>query(ms, parameterObject, rowBounds, resultHandler, key, boundSql);
106   }
107 
108   public List<BatchResult> flushStatements() throws SQLException {
109     return delegate.flushStatements();
110   }
111 
112   public void commit(boolean required) throws SQLException {
113     delegate.commit(required);
114     tcm.commit();
115     dirty = false;
116   }
117 
118   public void rollback(boolean required) throws SQLException {
119     try {
120       delegate.rollback(required);
121       dirty = false;
122     } finally {
123       if (required) {
124         tcm.rollback();
125       }
126     }
127   }
128 
129   private void ensureNoOutParams(MappedStatement ms, Object parameter, BoundSql boundSql) {
130     if (ms.getStatementType() == StatementType.CALLABLE) {
131       for (ParameterMapping parameterMapping : boundSql.getParameterMappings()) {
132         if (parameterMapping.getMode() != ParameterMode.IN) {
133           throw new ExecutorException("Caching stored procedures with OUT params is not supported.  Please configure useCache=false in " + ms.getId() + " statement.");
134         }
135       }
136     }
137   }
138 
139   public CacheKey createCacheKey(MappedStatement ms, Object parameterObject, RowBounds rowBounds, BoundSql boundSql) {
140     return delegate.createCacheKey(ms, parameterObject, rowBounds, boundSql);
141   }
142 
143   public boolean isCached(MappedStatement ms, CacheKey key) {
144     throw new UnsupportedOperationException("The CachingExecutor should not be used by result loaders and thus isCached() should never be called.");
145   }
146 
147   public void deferLoad(MappedStatement ms, MetaObject resultObject, String property, CacheKey key, Class<?> targetType) {
148     throw new UnsupportedOperationException("The CachingExecutor should not be used by result loaders and thus deferLoad() should never be called.");
149   }
150 
151   public void clearLocalCache() {
152     delegate.clearLocalCache();
153   }
154 
155   private void flushCacheIfRequired(MappedStatement ms) {
156     Cache cache = ms.getCache();
157     if (cache != null && ms.isFlushCacheRequired()) {
158       dirty = true; // issue #524. Disable using cached data for this session
159       tcm.clear(cache);
160     }
161   }
