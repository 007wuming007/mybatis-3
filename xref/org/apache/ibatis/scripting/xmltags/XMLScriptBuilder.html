54    public XMLScriptBuilder(Configuration configuration, String context, Class<?> parameterType) {
55      super(configuration);
56      XPathParser parser = new XPathParser(context, false, configuration.getVariables(), new XMLMapperEntityResolver());
57      this.context = parser.evalNode("/script");
58      this.parameterType = parameterType;
59    }

61    public SqlSource parseScriptNode() {
62      List<SqlNode> contents = parseDynamicTags(context);
63      MixedSqlNode rootSqlNode = new MixedSqlNode(contents);
64      SqlSource sqlSource = null;
65      if (isDynamic) {
66        sqlSource = new DynamicSqlSource(configuration, rootSqlNode);
67      } else {
68        sqlSource = new RawSqlSource(configuration, context, parameterType);
69      }
70      return sqlSource;
71    }

73    private List<SqlNode> parseDynamicTags(XNode node) {
74      List<SqlNode> contents = new ArrayList<SqlNode>();
75      NodeList children = node.getNode().getChildNodes();
76      for (int i = 0; i < children.getLength(); i++) {
77        XNode child = node.newXNode(children.item(i));
78        String nodeName = child.getNode().getNodeName();
79        if (child.getNode().getNodeType() == Node.CDATA_SECTION_NODE
80            || child.getNode().getNodeType() == Node.TEXT_NODE) {
81          String data = child.getStringBody("");
82          TextSqlNode textSqlNode = new TextSqlNode(data);
83          if (textSqlNode.isDynamic()) {
84            contents.add(textSqlNode);
85            isDynamic = true;
86          } else {
87            contents.add(new StaticTextSqlNode(data));
88          }
89        } else if (child.getNode().getNodeType() == Node.ELEMENT_NODE && !"selectKey".equals(nodeName)) { // issue #628
90          NodeHandler handler = nodeHandlers.get(nodeName);
91          if (handler == null) {
92            throw new BuilderException("Unknown element <" + nodeName + "> in SQL statement.");
93          }
94          handler.handleNode(child, contents);
95          isDynamic = true;
96        }
97      }
98      return contents;
99    }

132       List<SqlNode> contents = parseDynamicTags(nodeToHandle);

194   private class ChooseHandler implements NodeHandler {
195     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
196       List<SqlNode> whenSqlNodes = new ArrayList<SqlNode>();
197       List<SqlNode> otherwiseSqlNodes = new ArrayList<SqlNode>();
198       handleWhenOtherwiseNodes(nodeToHandle, whenSqlNodes, otherwiseSqlNodes);
199       SqlNode defaultSqlNode = getDefaultSqlNode(otherwiseSqlNodes);
200       ChooseSqlNode chooseSqlNode = new ChooseSqlNode(whenSqlNodes, defaultSqlNode);
201       targetContents.add(chooseSqlNode);
202     }
203 
204     private void handleWhenOtherwiseNodes(XNode chooseSqlNode, List<SqlNode> ifSqlNodes, List<SqlNode> defaultSqlNodes) {
205       List<XNode> children = chooseSqlNode.getChildren();
206       for (XNode child : children) {
207         String nodeName = child.getNode().getNodeName();
208         NodeHandler handler = nodeHandlers.get(nodeName);
209         if (handler instanceof IfHandler) {
210           handler.handleNode(child, ifSqlNodes);
211         } else if (handler instanceof OtherwiseHandler) {
212           handler.handleNode(child, defaultSqlNodes);
213         }
214       }
215     }
216 
217     private SqlNode getDefaultSqlNode(List<SqlNode> defaultSqlNodes) {
218       SqlNode defaultSqlNode = null;
219       if (defaultSqlNodes.size() == 1) {
220         defaultSqlNode = defaultSqlNodes.get(0);
221       } else if (defaultSqlNodes.size() > 1) {
222         throw new BuilderException("Too many default (otherwise) elements in choose statement.");
223       }
224       return defaultSqlNode;
225     }
