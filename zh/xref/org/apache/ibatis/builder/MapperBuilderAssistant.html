69    public void setCurrentNamespace(String currentNamespace) {
70      if (currentNamespace == null) {
71        throw new BuilderException("The mapper element requires a namespace attribute to be specified.");
72      }
73  
74      if (this.currentNamespace != null && !this.currentNamespace.equals(currentNamespace)) {
75        throw new BuilderException("Wrong namespace. Expected '"
76            + this.currentNamespace + "' but found '" + currentNamespace + "'.");
77      }
78  
79      this.currentNamespace = currentNamespace;
80    }

82    public String applyCurrentNamespace(String base, boolean isReference) {
83      if (base == null) return null;
84      if (isReference) {
85        // is it qualified with any namespace yet?
86        if (base.contains(".")) return base;
87      } else {
88        // is it qualified with this namespace yet?
89        if (base.startsWith(currentNamespace + ".")) return base;
90        if (base.contains(".")) throw new BuilderException("Dots are not allowed in element names, please remove it from " + base);
91      }
92      return currentNamespace + "." + base;
93    }

95    public Cache useCacheRef(String namespace) {
96      if (namespace == null) {
97        throw new BuilderException("cache-ref element requires a namespace attribute.");
98      }
99      try {
100       unresolvedCacheRef = true;
101       Cache cache = configuration.getCache(namespace);
102       if (cache == null) {
103         throw new IncompleteElementException("No cache for namespace '" + namespace + "' could be found.");
104       }
105       currentCache = cache;
106       unresolvedCacheRef = false;
107       return cache;
108     } catch (IllegalArgumentException e) {
109       throw new IncompleteElementException("No cache for namespace '" + namespace + "' could be found.", e);
110     }
111   }

113   public Cache useNewCache(Class<? extends Cache> typeClass,
114       Class<? extends Cache> evictionClass,
115       Long flushInterval,
116       Integer size,
117       boolean readWrite,
118       Properties props) {
119     typeClass = valueOrDefault(typeClass, PerpetualCache.class);
120     evictionClass = valueOrDefault(evictionClass, LruCache.class);
121     Cache cache = new CacheBuilder(currentNamespace)
122         .implementation(typeClass)
123         .addDecorator(evictionClass)
124         .clearInterval(flushInterval)
125         .size(size)
126         .readWrite(readWrite)
127         .properties(props)
128         .build();
129     configuration.addCache(cache);
130     currentCache = cache;
131     return cache;
132   }

166   public ResultMap addResultMap(
167       String id,
168       Class<?> type,
169       String extend,
170       Discriminator discriminator,
171       List<ResultMapping> resultMappings,
172       Boolean autoMapping) {
173     id = applyCurrentNamespace(id, false);
174     extend = applyCurrentNamespace(extend, true);
175 
176     ResultMap.Builder resultMapBuilder = new ResultMap.Builder(configuration, id, type, resultMappings, autoMapping);
177     if (extend != null) {
178       if (!configuration.hasResultMap(extend)) {
179         throw new IncompleteElementException("Could not find a parent resultmap with id '" + extend + "'");
180       }
181       ResultMap resultMap = configuration.getResultMap(extend);
182       List<ResultMapping> extendedResultMappings = new ArrayList<ResultMapping>(resultMap.getResultMappings());
183       extendedResultMappings.removeAll(resultMappings);
184       // Remove parent constructor if this resultMap declares a constructor.
185       boolean declaresConstructor = false;
186       for (ResultMapping resultMapping : resultMappings) {
187         if (resultMapping.getFlags().contains(ResultFlag.CONSTRUCTOR)) {
188           declaresConstructor = true;
189           break;
190         }
191       }
192       if (declaresConstructor) {
193         Iterator<ResultMapping> extendedResultMappingsIter = extendedResultMappings.iterator();
194         while (extendedResultMappingsIter.hasNext()) {
195           if (extendedResultMappingsIter.next().getFlags().contains(ResultFlag.CONSTRUCTOR)) {
196             extendedResultMappingsIter.remove();
197           }
198         }
199       }
200       resultMappings.addAll(extendedResultMappings);
201     }
202     resultMapBuilder.discriminator(discriminator);
203     ResultMap resultMap = resultMapBuilder.build();
204     configuration.addResultMap(resultMap);
205     return resultMap;
206   }

235   public Discriminator buildDiscriminator(
236       Class<?> resultType,
237       String column,
238       Class<?> javaType,
239       JdbcType jdbcType,
240       Class<? extends TypeHandler<?>> typeHandler,
241       Map<String, String> discriminatorMap) {
242     ResultMapping resultMapping = assembleResultMapping(
243         resultType,
244         null,
245         column,
246         javaType,
247         jdbcType,
248         null,
249         null,
250         null,
251         null,
252         typeHandler,
253         new ArrayList<ResultFlag>());
254     Map<String, String> namespaceDiscriminatorMap = new HashMap<String, String>();
255     for (Map.Entry<String, String> e : discriminatorMap.entrySet()) {
256       String resultMap = e.getValue();
257       resultMap = applyCurrentNamespace(resultMap, true);
258       namespaceDiscriminatorMap.put(e.getKey(), resultMap);
259     }
260     Discriminator.Builder discriminatorBuilder = new Discriminator.Builder(configuration, resultMapping, namespaceDiscriminatorMap);
261     return discriminatorBuilder.build();
262   }

264   public MappedStatement addMappedStatement(
265       String id,
266       SqlSource sqlSource,
267       StatementType statementType,
268       SqlCommandType sqlCommandType,
269       Integer fetchSize,
270       Integer timeout,
271       String parameterMap,
272       Class<?> parameterType,
273       String resultMap,
274       Class<?> resultType,
275       ResultSetType resultSetType,
276       boolean flushCache,
277       boolean useCache,
278       boolean resultOrdered,
279       KeyGenerator keyGenerator,
280       String keyProperty,
281       String keyColumn,
282       String databaseId,
283       LanguageDriver lang) {
284     
285     if (unresolvedCacheRef) throw new IncompleteElementException("Cache-ref not yet resolved");
286     
287     id = applyCurrentNamespace(id, false);
288     boolean isSelect = sqlCommandType == SqlCommandType.SELECT;
289 
290     MappedStatement.Builder statementBuilder = new MappedStatement.Builder(configuration, id, sqlSource, sqlCommandType);
291     statementBuilder.resource(resource);
292     statementBuilder.fetchSize(fetchSize);
293     statementBuilder.statementType(statementType);
294     statementBuilder.keyGenerator(keyGenerator);
295     statementBuilder.keyProperty(keyProperty);
296     statementBuilder.keyColumn(keyColumn);
297     statementBuilder.databaseId(databaseId);
298     statementBuilder.lang(lang);
299     statementBuilder.resultOrdered(resultOrdered);
300     setStatementTimeout(timeout, statementBuilder);
301 
302     setStatementParameterMap(parameterMap, parameterType, statementBuilder);
303     setStatementResultMap(resultMap, resultType, resultSetType, statementBuilder);
304     setStatementCache(isSelect, flushCache, useCache, currentCache, statementBuilder);
305 
306     MappedStatement statement = statementBuilder.build();
307     configuration.addMappedStatement(statement);
308     return statement;
309   }

328   private void setStatementParameterMap(
329       String parameterMap,
330       Class<?> parameterTypeClass,
331       MappedStatement.Builder statementBuilder) {
332     parameterMap = applyCurrentNamespace(parameterMap, true);
333 
334     if (parameterMap != null) {
335       try {
336         statementBuilder.parameterMap(configuration.getParameterMap(parameterMap));
337       } catch (IllegalArgumentException e) {
338         throw new IncompleteElementException("Could not find parameter map " + parameterMap, e);
339       }
340     } else if (parameterTypeClass != null) {
341       List<ParameterMapping> parameterMappings = new ArrayList<ParameterMapping>();
342       ParameterMap.Builder inlineParameterMapBuilder = new ParameterMap.Builder(
343           configuration,
344           statementBuilder.id() + "-Inline",
345           parameterTypeClass,
346           parameterMappings);
347       statementBuilder.parameterMap(inlineParameterMapBuilder.build());
348     }

351   private void setStatementResultMap(
352       String resultMap,
353       Class<?> resultType,
354       ResultSetType resultSetType,
355       MappedStatement.Builder statementBuilder) {
356     resultMap = applyCurrentNamespace(resultMap, true);
357 
358     List<ResultMap> resultMaps = new ArrayList<ResultMap>();
359     if (resultMap != null) {
360       String[] resultMapNames = resultMap.split(",");
361       for (String resultMapName : resultMapNames) {
362         try {
363           resultMaps.add(configuration.getResultMap(resultMapName.trim()));
364         } catch (IllegalArgumentException e) {
365           throw new IncompleteElementException("Could not find result map " + resultMapName, e);
366         }
367       }
368     } else if (resultType != null) {
369       ResultMap.Builder inlineResultMapBuilder = new ResultMap.Builder(
370           configuration,
371           statementBuilder.id() + "-Inline",
372           resultType,
373           new ArrayList<ResultMapping>(),
374           null);
375       resultMaps.add(inlineResultMapBuilder.build());
376     }
377     statementBuilder.resultMaps(resultMaps);
378 
379     statementBuilder.resultSetType(resultSetType);
380   }

389   private ResultMapping assembleResultMapping(
390       Class<?> resultType,
391       String property,
392       String column,
393       Class<?> javaType,
394       JdbcType jdbcType,
395       String nestedSelect,
396       String nestedResultMap,
397       String notNullColumn,
398       String columnPrefix,
399       Class<? extends TypeHandler<?>> typeHandler,
400       List<ResultFlag> flags) {
401     Class<?> javaTypeClass = resolveResultJavaType(resultType, property, javaType);
402     TypeHandler<?> typeHandlerInstance = resolveTypeHandler(javaTypeClass, typeHandler);
403     List<ResultMapping> composites = parseCompositeColumnName(column);
404     if (composites.size() > 0) column = null;
405     ResultMapping.Builder builder = new ResultMapping.Builder(configuration, property, column, javaTypeClass);
406     builder.jdbcType(jdbcType);
407     builder.nestedQueryId(applyCurrentNamespace(nestedSelect, true));
408     builder.nestedResultMapId(applyCurrentNamespace(nestedResultMap, true));
409     builder.typeHandler(typeHandlerInstance);
410     builder.flags(flags == null ? new ArrayList<ResultFlag>() : flags);
411     builder.composites(composites);
412     builder.notNullColumns(parseMultipleColumnNames(notNullColumn));
413     builder.columnPrefix(columnPrefix);
414     return builder.build();
415   }

417   private Set<String> parseMultipleColumnNames(String columnName) {
418     Set<String> columns = new HashSet<String>();
419     if (columnName != null) {
420       if (columnName.indexOf(',') > -1) {
421         StringTokenizer parser = new StringTokenizer(columnName, "{}, ", false);
422         while (parser.hasMoreTokens()) {
423           String column = parser.nextToken();
424           columns.add(column);
425         }
426       } else {
427         columns.add(columnName);
428       }
429     }
430     return columns;
431   }
432 
433   private List<ResultMapping> parseCompositeColumnName(String columnName) {
434     List<ResultMapping> composites = new ArrayList<ResultMapping>();
435     if (columnName != null && (columnName.indexOf('=') > -1 || columnName.indexOf(',') > -1)) {
436       StringTokenizer parser = new StringTokenizer(columnName, "{}=, ", false);
437       while (parser.hasMoreTokens()) {
438         String property = parser.nextToken();
439         String column = parser.nextToken();
440         ResultMapping.Builder complexBuilder = new ResultMapping.Builder(configuration, property, column, configuration.getTypeHandlerRegistry().getUnknownTypeHandler());
441         composites.add(complexBuilder.build());
442       }
443     }
444     return composites;
445   }

447   private Class<?> resolveResultJavaType(Class<?> resultType, String property, Class<?> javaType) {
448     if (javaType == null && property != null) {
449       try {
450         MetaClass metaResultType = MetaClass.forClass(resultType);
451         javaType = metaResultType.getSetterType(property);
452       } catch (Exception e) {
453         //ignore, following null check statement will deal with the situation
454       }
455     }
456     if (javaType == null) {
457       javaType = Object.class;
458     }
459     return javaType;
460   }

462   private Class<?> resolveParameterJavaType(Class<?> resultType, String property, Class<?> javaType, JdbcType jdbcType) {
463     if (javaType == null) {
464       if (JdbcType.CURSOR.equals(jdbcType)) {
465         javaType = java.sql.ResultSet.class;
466       } else if (Map.class.isAssignableFrom(resultType)) {
467         javaType = Object.class;
468       } else {
469         MetaClass metaResultType = MetaClass.forClass(resultType);
470         javaType = metaResultType.getGetterType(property);
471       }
472     }
473     if (javaType == null) {
474       javaType = Object.class;
475     }
476     return javaType;
477   }
