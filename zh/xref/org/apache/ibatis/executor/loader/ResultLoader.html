16  package org.apache.ibatis.executor.loader;
17  
18  import java.sql.SQLException;
19  import java.util.List;
20  
21  import javax.sql.DataSource;
22  
23  import org.apache.ibatis.cache.CacheKey;
24  import org.apache.ibatis.executor.Executor;
25  import org.apache.ibatis.executor.ExecutorException;
26  import org.apache.ibatis.executor.ResultExtractor;
27  import org.apache.ibatis.mapping.BoundSql;
28  import org.apache.ibatis.mapping.Environment;
29  import org.apache.ibatis.mapping.MappedStatement;
30  import org.apache.ibatis.reflection.factory.ObjectFactory;
31  import org.apache.ibatis.session.Configuration;
32  import org.apache.ibatis.session.ExecutorType;
33  import org.apache.ibatis.session.RowBounds;
34  import org.apache.ibatis.transaction.Transaction;
35  import org.apache.ibatis.transaction.TransactionFactory;
36  
37  public class ResultLoader {
38  
39    protected final Configuration configuration;
40    protected final Executor executor;
41    protected final MappedStatement mappedStatement;
42    protected final Object parameterObject;
43    protected final Class<?> targetType;
44    protected final ObjectFactory objectFactory;
45    protected final CacheKey cacheKey;
46    protected final BoundSql boundSql;
47    protected final ResultExtractor resultExtractor;
48    
49    protected boolean loaded;
50    protected Object resultObject;
51    
52    public ResultLoader(Configuration config, Executor executor, MappedStatement mappedStatement, Object parameterObject, Class<?> targetType, CacheKey cacheKey, BoundSql boundSql) {
53      this.configuration = config;
54      this.executor = executor;
55      this.mappedStatement = mappedStatement;
56      this.parameterObject = parameterObject;
57      this.targetType = targetType;
58      this.objectFactory = configuration.getObjectFactory();
59      this.cacheKey = cacheKey;
60      this.boundSql = boundSql;
61      this.resultExtractor = new ResultExtractor(configuration, objectFactory);
62    }
63  
64    public Object loadResult() throws SQLException {
65      List<Object> list = selectList();
66      resultObject = resultExtractor.extractObjectFromList(list, targetType);
67      return resultObject;
68    }
69  
70    private <E> List<E> selectList() throws SQLException {
71      Executor localExecutor = executor;
72      if (localExecutor.isClosed()) {
73        localExecutor = newExecutor();
74      }
75      try {
76        return localExecutor.<E> query(mappedStatement, parameterObject, RowBounds.DEFAULT, Executor.NO_RESULT_HANDLER, cacheKey, boundSql);
77      } finally {
78        if (localExecutor != executor) {
79          localExecutor.close(false);
80        }
81      }
82    }
83  
84    private Executor newExecutor() throws SQLException {
85      final Environment environment = configuration.getEnvironment();
86      if (environment == null) throw new ExecutorException("ResultLoader could not load lazily.  Environment was not configured.");
87      final DataSource ds = environment.getDataSource();
88      if (ds == null) throw new ExecutorException("ResultLoader could not load lazily.  DataSource was not configured.");
89      final TransactionFactory transactionFactory = environment.getTransactionFactory();
90      final Transaction tx = transactionFactory.newTransaction(ds, null, false);
91      return configuration.newExecutor(tx, ExecutorType.SIMPLE);
92    }
93  
94    public boolean wasNull() {
95      return resultObject == null;
96    }

98  }
