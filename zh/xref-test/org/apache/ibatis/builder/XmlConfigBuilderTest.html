53    public static class EnumOrderTypeHandler<E extends Enum<E>> extends BaseTypeHandler<E> {
54  
55      private E[] constants;
56  
57      public EnumOrderTypeHandler(Class<E> javaType) {
58        constants = javaType.getEnumConstants();
59      }
60  
61      @Override
62      public void setNonNullParameter(PreparedStatement ps, int i, E parameter, JdbcType jdbcType) throws SQLException {
63        ps.setInt(i, parameter.ordinal() + 1); // 0 means NULL so add +1
64      }
65  
66      @Override
67      public E getNullableResult(ResultSet rs, String columnName) throws SQLException {
68        int index = rs.getInt(columnName) - 1;
69        return index < 0 ? null : constants[index];
70      }
71  
72      @Override
73      public E getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
74        int index = rs.getInt(rs.getInt(columnIndex)) - 1;
75        return index < 0 ? null : constants[index];
76      }
77  
78      @Override
79      public E getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
80        int index = cs.getInt(columnIndex) - 1;
81        return index < 0 ? null : constants[index];
82      }
83    }

87      final String MAPPER_CONFIG = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n"
88          + "<!DOCTYPE configuration PUBLIC \"-//mybatis.org//DTD Config 3.0//EN\" \"http://mybatis.org/dtd/mybatis-3-config.dtd\">\n" 
89          + "<configuration>\n" 
90          + "  <typeHandlers>\n"
91          + "    <typeHandler javaType=\"org.apache.ibatis.builder.XmlConfigBuilderTest$MyEnum\"\n"
92          + "      handler=\"org.apache.ibatis.builder.XmlConfigBuilderTest$EnumOrderTypeHandler\"/>\n" 
93          + "  </typeHandlers>\n" 
94          + "</configuration>\n";
95  
96      XMLConfigBuilder builder = new XMLConfigBuilder(new StringReader(MAPPER_CONFIG));
97      builder.parse();
98  
99      TypeHandlerRegistry typeHandlerRegistry = builder.getConfiguration().getTypeHandlerRegistry();
100     TypeHandler<MyEnum> typeHandler = typeHandlerRegistry.getTypeHandler(MyEnum.class);
101 
102     assertTrue(typeHandler instanceof EnumOrderTypeHandler);
103     assertArrayEquals(MyEnum.values(), ((EnumOrderTypeHandler) typeHandler).constants);
104   }
