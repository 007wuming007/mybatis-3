49    private void testCrossReference(SqlSessionFactory sqlSessionFactory) throws Exception {
50      SqlSession sqlSession = sqlSessionFactory.openSession();
51      try {
52    	  ParameterMapReferencePersonMapper personMapper = sqlSession.getMapper(ParameterMapReferencePersonMapper.class);
53    	  Person parameter = new Person();
54    	  parameter.setId(1);
55        Person person = personMapper.select(parameter);
56        assertEquals((Integer)1, person.getId());
57  
58      } finally {
59        sqlSession.close();
60      }
61    }

63    private SqlSessionFactory getSqlSessionFactoryXmlConfig() throws Exception {
64      Reader configReader = Resources
65          .getResourceAsReader("org/apache/ibatis/submitted/xml_external_ref/ParameterMapReferenceMapperConfig.xml");
66      SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(configReader);
67      configReader.close();
68  
69      Connection conn = sqlSessionFactory.getConfiguration().getEnvironment().getDataSource().getConnection();
70      initDb(conn);
71  
72      return sqlSessionFactory;
73    }
74  
75    private SqlSessionFactory getSqlSessionFactoryJavaConfig() throws Exception {
76      Class.forName("org.hsqldb.jdbcDriver");
77      Connection c = DriverManager.getConnection("jdbc:hsqldb:mem:xmlextref", "sa", "");
78      initDb(c);
79  
80      Configuration configuration = new Configuration();
81      Environment environment = new Environment("development", new JdbcTransactionFactory(), new UnpooledDataSource(
82          "org.hsqldb.jdbcDriver", "jdbc:hsqldb:mem:xmlextref", null));
83      configuration.setEnvironment(environment);
84  
85      configuration.addMapper(ParameterMapReferencePersonMapper.class);
86      configuration.addMapper(ParameterMapReferencePetMapper.class);
87  
88      return new SqlSessionFactoryBuilder().build(configuration);
89    }
90  
91    private static void initDb(Connection conn) throws IOException, SQLException {
92      try {
93        Reader scriptReader = Resources.getResourceAsReader("org/apache/ibatis/submitted/xml_external_ref/CreateDB.sql");
94        ScriptRunner runner = new ScriptRunner(conn);
95        runner.setLogWriter(null);
96        runner.setErrorLogWriter(null);
97        runner.runScript(scriptReader);
98        conn.commit();
99        scriptReader.close();
100     } finally {
101       if (conn != null) {
102         conn.close();
103       }
104     }
105   }
