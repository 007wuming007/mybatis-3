182   protected byte[] serialize(Serializable value) throws Exception {
183     ByteArrayOutputStream bos = new ByteArrayOutputStream();
184     ObjectOutputStream oos = new ObjectOutputStream(bos);
185     oos.writeObject(value);
186     oos.flush();
187     oos.close();
188     return bos.toByteArray();
189   }
190 
191   protected Serializable deserialize(byte[] value) throws Exception {
192     ByteArrayInputStream bis = new ByteArrayInputStream(value);
193     ObjectInputStream ois = new ObjectInputStream(bis);
194     Serializable result = (Serializable) ois.readObject();
195     ois.close();
196     return result;
197   }
