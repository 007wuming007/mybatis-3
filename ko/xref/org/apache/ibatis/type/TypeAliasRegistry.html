100   @SuppressWarnings("unchecked")
101   // throws class cast exception as well if types cannot be assigned
102   public <T> Class<T> resolveAlias(String string) {
103     try {
104       if (string == null) return null;
105       String key = string.toLowerCase(Locale.ENGLISH); // issue #748
106       Class<T> value;
107       if (TYPE_ALIASES.containsKey(key)) {
108         value = (Class<T>) TYPE_ALIASES.get(key);
109       } else {
110         value = (Class<T>) Resources.classForName(string);
111       }
112       return value;
113     } catch (ClassNotFoundException e) {
114       throw new TypeException("Could not resolve type alias '" + string + "'.  Cause: " + e, e);
115     }
116   }

122   public void registerAliases(String packageName, Class<?> superType){
123     ResolverUtil<Class<?>> resolverUtil = new ResolverUtil<Class<?>>();
124     resolverUtil.find(new ResolverUtil.IsA(superType), packageName);
125     Set<Class<? extends Class<?>>> typeSet = resolverUtil.getClasses();
126     for(Class<?> type : typeSet){
127       // Ignore inner classes and interfaces (including package-info.java)
128       // Skip also inner classes. See issue #6
129       if (!type.isAnonymousClass() && !type.isInterface() && !type.isMemberClass()) {
130         registerAlias(type);
131       }
132     }
133   }

135   public void registerAlias(Class<?> type) {
136     String alias = type.getSimpleName();
137     Alias aliasAnnotation = type.getAnnotation(Alias.class);
138     if (aliasAnnotation != null) {
139       alias = aliasAnnotation.value();
140     } 
141     registerAlias(alias, type);
142   }
143 
144   public void registerAlias(String alias, Class<?> value) {
145     if (alias == null) throw new TypeException("The parameter alias cannot be null");
146     String key = alias.toLowerCase(Locale.ENGLISH); // issue #748
147     if (TYPE_ALIASES.containsKey(key) && TYPE_ALIASES.get(key) != null && !TYPE_ALIASES.get(key).equals(value)) {
148       throw new TypeException("The alias '" + alias + "' is already mapped to the value '" + TYPE_ALIASES.get(key).getName() + "'.");
149     }
150     TYPE_ALIASES.put(key, value);
151   }
152 
153   public void registerAlias(String alias, String value) {
154     try {
155       registerAlias(alias, Resources.classForName(value));
156     } catch (ClassNotFoundException e) {
157       throw new TypeException("Error registering type alias "+alias+" for "+value+". Cause: " + e, e);
158     }
159   }
