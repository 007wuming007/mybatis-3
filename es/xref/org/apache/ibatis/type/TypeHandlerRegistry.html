177   private <T> TypeHandler<T> getTypeHandler(Type type, JdbcType jdbcType) {
178     Map<JdbcType, TypeHandler<?>> jdbcHandlerMap = TYPE_HANDLER_MAP.get(type);
179     TypeHandler<?> handler = null;
180     if (jdbcHandlerMap != null) {
181       handler = jdbcHandlerMap.get(jdbcType);
182       if (handler == null) {
183         handler = jdbcHandlerMap.get(null);
184       }
185     }
186     if (handler == null && type != null && type instanceof Class && Enum.class.isAssignableFrom((Class<?>) type)) {
187       handler = new EnumTypeHandler((Class<?>) type);
188     }
189     @SuppressWarnings("unchecked")
190     // type drives generics here
191     TypeHandler<T> returned = (TypeHandler<T>) handler;
192     return returned;
193   }

209   @SuppressWarnings("unchecked")
210   public <T> void register(TypeHandler<T> typeHandler) {
211     boolean mappedTypeFound = false;
212     MappedTypes mappedTypes = typeHandler.getClass().getAnnotation(MappedTypes.class);
213     if (mappedTypes != null) {
214       for (Class<?> handledType : mappedTypes.value()) {
215         register(handledType, typeHandler);
216         mappedTypeFound = true;
217       }
218     }
219     // @since 3.1.0 - try to auto-discover the mapped type
220     if (!mappedTypeFound && typeHandler instanceof TypeReference) {
221       try {
222         TypeReference<T> typeReference = (TypeReference<T>) typeHandler;
223         register(typeReference.getRawType(), typeHandler);
224         mappedTypeFound = true;
225       } catch (Throwable t) {
226         // maybe users define the TypeReference with a different type and are not assignable, so just ignore it
227       }
228     }
229     if (!mappedTypeFound) {
230       register((Class<T>) null, typeHandler);
231     }
232   }

240   private <T> void register(Type javaType, TypeHandler<? extends T> typeHandler) {
241     MappedJdbcTypes mappedJdbcTypes = typeHandler.getClass().getAnnotation(MappedJdbcTypes.class);
242     if (mappedJdbcTypes != null) {
243       for (JdbcType handledJdbcType : mappedJdbcTypes.value()) {
244         register(javaType, handledJdbcType, typeHandler);
245       }
246       if (mappedJdbcTypes.includeNullJdbcType()) {
247         register(javaType, null, typeHandler);
248       }
249     } else {
250       register(javaType, null, typeHandler);
251     }
252   }

264   private void register(Type javaType, JdbcType jdbcType, TypeHandler<?> handler) {
265     if (javaType != null) {
266       Map<JdbcType, TypeHandler<?>> map = TYPE_HANDLER_MAP.get(javaType);
267       if (map == null) {
268         map = new HashMap<JdbcType, TypeHandler<?>>();
269         TYPE_HANDLER_MAP.put(javaType, map);
270       }
271       map.put(jdbcType, handler);
272       if (reversePrimitiveMap.containsKey(javaType)) {
273         register(reversePrimitiveMap.get(javaType), jdbcType, handler);
274       }
275     }
276     ALL_TYPE_HANDLERS_MAP.put(handler.getClass(), handler);
277   }

313   @SuppressWarnings("unchecked")
314   public <T> TypeHandler<T> getInstance(Class<?> javaTypeClass, Class<?> typeHandlerClass) {
315     if (javaTypeClass != null) {
316       try {
317         Constructor<?> c = typeHandlerClass.getConstructor(Class.class);
318         return (TypeHandler<T>) c.newInstance(javaTypeClass);
319       } catch (NoSuchMethodException ignored) {
320         // ignored
321       } catch (Exception e) {
322         throw new RuntimeException("Failed invoking constructor for handler " + typeHandlerClass, e);
323       }
324     }
325     try {
326       Constructor<?> c = typeHandlerClass.getConstructor();
327       return (TypeHandler<T>) c.newInstance();
328     } catch (Exception e) {
329       throw new RuntimeException("Unable to find a usable constructor for " + typeHandlerClass, e);
330     }
331   }

335   public void register(String packageName) {
336     ResolverUtil<Class<?>> resolverUtil = new ResolverUtil<Class<?>>();
337     resolverUtil.find(new ResolverUtil.IsA(TypeHandler.class), packageName);
338     Set<Class<? extends Class<?>>> handlerSet = resolverUtil.getClasses();
339     for (Class<?> type : handlerSet) {
340       //Ignore inner classes and interfaces (including package-info.java) and abstract classes
341       if (!type.isAnonymousClass() && !type.isInterface() && !Modifier.isAbstract(type.getModifiers())) {
342         register(type);
343       }
344     }
