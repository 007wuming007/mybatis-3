16  package org.apache.ibatis.logging.jdbc;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.Method;
20  import java.lang.reflect.Proxy;
21  import java.sql.ResultSet;
22  import java.sql.ResultSetMetaData;
23  import java.sql.SQLException;
24  import java.sql.Types;
25  import java.util.HashSet;
26  import java.util.Set;
27  
28  import org.apache.ibatis.logging.Log;
29  import org.apache.ibatis.reflection.ExceptionUtil;
30  
31  /**
32   * ResultSet proxy to add logging
33   * 
34   * @author Clinton Begin
35   * @author Eduardo Macarron
36   * 
37   */
38  public final class ResultSetLogger extends BaseJdbcLogger implements InvocationHandler {
39  
40    private static Set<Integer> BLOB_TYPES = new HashSet<Integer>();
41    private boolean first = true;
42    private int rows = 0;
43    private ResultSet rs;
44    private Set<Integer> blobColumns = new HashSet<Integer>();
45  
46    static {
47      BLOB_TYPES.add(Types.BINARY);
48      BLOB_TYPES.add(Types.BLOB);
49      BLOB_TYPES.add(Types.CLOB);
50      BLOB_TYPES.add(Types.LONGNVARCHAR);
51      BLOB_TYPES.add(Types.LONGVARBINARY);
52      BLOB_TYPES.add(Types.LONGVARCHAR);
53      BLOB_TYPES.add(Types.NCLOB);
54      BLOB_TYPES.add(Types.VARBINARY);
55    }
56    
57    private ResultSetLogger(ResultSet rs, Log statementLog, int queryStack) {
58      super(statementLog, queryStack);
59      this.rs = rs;
60    }
61  
62    public Object invoke(Object proxy, Method method, Object[] params) throws Throwable {
63      try {
64        if (Object.class.equals(method.getDeclaringClass())) {
65          return method.invoke(this, params);
66        }    
67        Object o = method.invoke(rs, params);
68        if ("next".equals(method.getName())) {
69          if (((Boolean) o)) {
70            rows++;
71            if (isTraceEnabled()) {
72              ResultSetMetaData rsmd = rs.getMetaData();
73              final int columnCount = rsmd.getColumnCount();
74              if (first) {
75                first = false;
76                printColumnHeaders(rsmd, columnCount);
77              }
78              printColumnValues(columnCount);
79            }
80          } else {
81            debug("     Total: " + rows, false);
82          }
83        }
84        clearColumnInfo();
85        return o;
86      } catch (Throwable t) {
87        throw ExceptionUtil.unwrapThrowable(t);
88      }
89    }
90  
91    private void printColumnHeaders(ResultSetMetaData rsmd, int columnCount) throws SQLException {
92      StringBuilder row = new StringBuilder();
93      row.append("   Columns: ");
94      for (int i = 1; i <= columnCount; i++) {
95        if (BLOB_TYPES.contains(rsmd.getColumnType(i))) {
96          blobColumns.add(i);
97        }
98        String colname = rsmd.getColumnLabel(i);
99        row.append(colname);
100       if (i != columnCount) row.append(", ");
101     }
102     trace(row.toString(), false);
103   }
104 
105   private void printColumnValues(int columnCount) throws SQLException {
106     StringBuilder row = new StringBuilder();
107     row.append("       Row: ");
108     for (int i = 1; i <= columnCount; i++) {
109       String colname;
110       try {
111         if (blobColumns.contains(i)) {
112           colname = "<<BLOB>>";
113         } else {
114           colname = rs.getString(i);
115         }
116       } catch (SQLException e) {
117         // generally can't call getString() on a BLOB column
118         colname = "<<Cannot Display>>";
119       }
120       row.append(colname);
121       if (i != columnCount) row.append(", ");
122     }
123     trace(row.toString(), false);
124   }
125 
126   /*
127    * Creates a logging version of a ResultSet
128    *
129    * @param rs - the ResultSet to proxy
130    * @return - the ResultSet with logging
131    */
132   public static ResultSet newInstance(ResultSet rs, Log statementLog, int queryStack) {
133     InvocationHandler handler = new ResultSetLogger(rs, statementLog, queryStack);
134     ClassLoader cl = ResultSet.class.getClassLoader();
135     return (ResultSet) Proxy.newProxyInstance(cl, new Class[]{ResultSet.class}, handler);
136   }
137 
138   /*
139    * Get the wrapped result set
140    *
141    * @return the resultSet
142    */
143   public ResultSet getRs() {
144     return rs;
145   }
