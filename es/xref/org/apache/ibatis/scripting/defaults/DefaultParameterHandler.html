16  package org.apache.ibatis.scripting.defaults;
17  
18  import java.sql.PreparedStatement;
19  import java.sql.SQLException;
20  import java.util.List;
21  
22  import org.apache.ibatis.executor.ErrorContext;
23  import org.apache.ibatis.executor.parameter.ParameterHandler;
24  import org.apache.ibatis.mapping.BoundSql;
25  import org.apache.ibatis.mapping.MappedStatement;
26  import org.apache.ibatis.mapping.ParameterMapping;
27  import org.apache.ibatis.mapping.ParameterMode;
28  import org.apache.ibatis.reflection.MetaObject;
29  import org.apache.ibatis.session.Configuration;
30  import org.apache.ibatis.type.JdbcType;
31  import org.apache.ibatis.type.TypeHandler;
32  import org.apache.ibatis.type.TypeHandlerRegistry;
33  
34  public class DefaultParameterHandler implements ParameterHandler {
35  
36    private final TypeHandlerRegistry typeHandlerRegistry;
37  
38    private final MappedStatement mappedStatement;
39    private final Object parameterObject;
40    private BoundSql boundSql;
41    private Configuration configuration;
42  
43    public DefaultParameterHandler(MappedStatement mappedStatement, Object parameterObject, BoundSql boundSql) {
44      this.mappedStatement = mappedStatement;
45      this.configuration = mappedStatement.getConfiguration();
46      this.typeHandlerRegistry = mappedStatement.getConfiguration().getTypeHandlerRegistry();
47      this.parameterObject = parameterObject;
48      this.boundSql = boundSql;
49    }
50  
51    public Object getParameterObject() {
52      return parameterObject;
53    }
54  
55    public void setParameters(PreparedStatement ps) throws SQLException {
56      ErrorContext.instance().activity("setting parameters").object(mappedStatement.getParameterMap().getId());
57      List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
58      if (parameterMappings != null) {
59        MetaObject metaObject = parameterObject == null ? null : configuration.newMetaObject(parameterObject);
60        for (int i = 0; i < parameterMappings.size(); i++) {
61          ParameterMapping parameterMapping = parameterMappings.get(i);
62          if (parameterMapping.getMode() != ParameterMode.OUT) {
63            Object value;
64            String propertyName = parameterMapping.getProperty();
65            if (boundSql.hasAdditionalParameter(propertyName)) { // issue #448 ask first for additional params
66              value = boundSql.getAdditionalParameter(propertyName);
67            } else if (parameterObject == null) {
68              value = null;
69            } else if (typeHandlerRegistry.hasTypeHandler(parameterObject.getClass())) {
70              value = parameterObject;
71            } else {
72              value = metaObject == null ? null : metaObject.getValue(propertyName);
73            }
74            TypeHandler typeHandler = parameterMapping.getTypeHandler();
75            JdbcType jdbcType = parameterMapping.getJdbcType();
76            if (value == null && jdbcType == null) jdbcType = configuration.getJdbcTypeForNull();
77            typeHandler.setParameter(ps, i + 1, value, jdbcType);
78          }
79        }
80      }
81    }
82  
83  }
