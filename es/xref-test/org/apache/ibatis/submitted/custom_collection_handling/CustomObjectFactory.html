38      private <T> T instantiateClass(Class<T> type, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
39          try {
40              Constructor<T> constructor;
41              if (constructorArgTypes == null || constructorArgs == null) {
42                  constructor = type.getDeclaredConstructor();
43                  if (!constructor.isAccessible()) {
44                      constructor.setAccessible(true);
45                  }
46                  return constructor.newInstance();
47              }
48              constructor = type.getDeclaredConstructor(constructorArgTypes.toArray(new Class[constructorArgTypes.size()]));
49              if (!constructor.isAccessible()) {
50                  constructor.setAccessible(true);
51              }
52              return constructor.newInstance(constructorArgs.toArray(new Object[constructorArgs.size()]));
53          } catch (Exception e) {
54              StringBuilder argTypes = new StringBuilder();
55              if (constructorArgTypes != null) {
56                  for (Class<?> argType : constructorArgTypes) {
57                      argTypes.append(argType.getSimpleName());
58                      argTypes.append(",");
59                  }
60              }
61              StringBuilder argValues = new StringBuilder();
62              if (constructorArgs != null) {
63                  for (Object argValue : constructorArgs) {
64                      argValues.append(String.valueOf(argValue));
65                      argValues.append(",");
66                  }
67              }
68              throw new ReflectionException("Error instantiating " + type + " with invalid types (" + argTypes + ") or values (" + argValues + "). Cause: " + e, e);
69          }
70      }

72      private Class<?> resolveInterface(Class<?> type) {
73          Class<?> classToCreate;
74          if (type == List.class || type == Collection.class) {
75              classToCreate = LinkedList.class;
76          } else if (type == Map.class) {
77              classToCreate = LinkedHashMap.class;
78          } else if (type == SortedSet.class) { // issue #510 Collections Support
79              classToCreate = TreeSet.class;
80          } else if (type == Set.class) {
81              classToCreate = HashSet.class;
82          } else {
83              classToCreate = type;
84          }
85          return classToCreate;
86      }
