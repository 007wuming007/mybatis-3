471     @SelectProvider(type = ErrorSqlBuilder.class, method = "overload")
472     void methodOverload(String value);
473 
474     @SelectProvider(type = ErrorSqlBuilder.class, method = "invokeError")
475     void invokeError();
476 
477     @SelectProvider(type = ErrorSqlBuilder.class, method = "multipleProviderContext")
478     void multipleProviderContext();
479   }

504   public interface StaticMethodSqlProviderMapper {
505     @SelectProvider(type = SqlProvider.class, method = "noArgument")
506     int noArgument();
507 
508     @SelectProvider(type = SqlProvider.class, method = "oneArgument")
509     int oneArgument(Integer value);
510 
511     @SelectProvider(type = SqlProvider.class, method = "multipleArgument")
512     int multipleArgument(@Param("value1") Integer value1, @Param("value2") Integer value2);
513 
514     @SelectProvider(type = SqlProvider.class, method = "onlyProviderContext")
515     String onlyProviderContext();
516 
517     @SelectProvider(type = SqlProvider.class, method = "oneArgumentAndProviderContext")
518     String oneArgumentAndProviderContext(Integer value);
519 
520     @SuppressWarnings("unused")
521     class SqlProvider {
522 
523       public static String noArgument() {
524         return "SELECT 1 FROM INFORMATION_SCHEMA.SYSTEM_USERS";
525       }
526 
527       public static StringBuilder oneArgument(Integer value) {
528         return new StringBuilder().append("SELECT ").append(value)
529             .append(" FROM INFORMATION_SCHEMA.SYSTEM_USERS");
530       }
531 
532       public static CharSequence multipleArgument(@Param("value1") Integer value1,
533           @Param("value2") Integer value2) {
534         return "SELECT " + (value1 + value2) + " FROM INFORMATION_SCHEMA.SYSTEM_USERS";
535       }
536 
537       public static CharSequence onlyProviderContext(ProviderContext context) {
538         return new StringBuilder().append("SELECT '").append(context.getMapperMethod().getName())
539             .append("' FROM INFORMATION_SCHEMA.SYSTEM_USERS");
540       }
541 
542       public static String oneArgumentAndProviderContext(Integer value, ProviderContext context) {
543         return "SELECT '" + context.getMapperMethod().getName() + " " + value
544             + "' FROM INFORMATION_SCHEMA.SYSTEM_USERS";
545       }
546 
547     }

619           return "SELECT '" + context.getDatabaseId() + "' FROM INFORMATION_SCHEMA.SYSTEM_USERS";
620         } else {
621           return "SELECT '" + context.getDatabaseId() + "' FROM SYSIBM.SYSDUMMY1";
